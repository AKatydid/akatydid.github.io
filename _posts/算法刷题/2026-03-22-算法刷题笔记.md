#### 1.2.2 并查集（Union-Find）

<span id="note-200-Island_num"></span>

并查集一般用于**动态维护若干个不相交集合**，并支持高效的合并与连通性查询

```c++
class UnionFind{
public: 

    UnionFind(const vector<vector<char>>& grid){
        int m = grid.size();
        int n = grid[0].size();
    
        for (int i = 0; i < m; ++i){
            for (int j = 0; j < n; ++j){
                if (grid[i][j] == '1'){
                    this->m_cluster++;
                    this->m_parent.emplace_back(i * n + j);
                }else{
                    this->m_parent.emplace_back(-1);
                }
                this->m_rank.emplace_back(1);
            }
        }
    }

    int find(int x){
        if (m_parent[x] == -1) return -1;
        if (x != this->m_parent[x]){
            x = find(this->m_parent[x]);
        }
        return this->m_parent[x];
    }

    void unite(int a, int b){
        int a_parent = find(a);
        int b_parent = find(b);

        if (a_parent == -1 || b_parent == -1) return;
        if (a_parent == b_parent) return;

        int a_parent_rank = this->m_rank[a_parent];
        int b_parent_rank = this->m_rank[b_parent];

        // compare rank
        if (a_parent_rank < b_parent_rank){
            this->m_rank[b_parent]++;
            this->m_parent[a_parent] = b_parent;
        }else{
            this->m_rank[a_parent]++;
            this->m_parent[b_parent] = a_parent;
        }
        this->m_cluster--;
    }

    int get_cluster(){
        return this->m_cluster;
    }

private:
    vector<int> m_parent;
    vector<int> m_rank;
    int m_cluster = 0;
};

class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        UnionFind uf(grid);
        int m = grid.size();
        int n = grid[0].size();
        for(int i = 0; i < m; ++i){
            for (int j = 0; j < n; ++j){
                // up
                if (j-1 >= 0 && grid[i][j-1] == '1') uf.unite(i*n+j, i*n+j-1);
                // dowm
                if (j+1 < n && grid[i][j+1] == '1') uf.unite(i*n+j, i*n+j+1);
                // left
                if (i-1 >= 0 && grid[i-1][j] == '1') uf.unite(i*n+j, (i-1)*n+j);
                //right
                if (i+1 < m && grid[i+1][j] == '1') uf.unite(i*n+j, (i+1)*n+j);
            }
        }

        return uf.get_cluster();
    }
};
```

#### 1.3.1 LRU Cache
<span id="note-146-LRUCache"></span>

思路：双端队列（带空结点）+ Hash 表

```c++
struct LRUNode{
    LRUNode* pre;
    LRUNode* next;
    int key;
    int value;

    LRUNode(int k, int v): key(k), value(v), pre(nullptr), next(nullptr){}
};

class LRUCache {
public:
    LRUCache(int capacity) {
        this->head = new LRUNode(-1, -1);  // latest
        this->tail = new LRUNode(-1, -1);  // oldest
        this->head->next = this->tail;
        this->tail->pre = this->head;
        this->m_size = 0;
        this->m_capacity = capacity;
    }
    
    int get(int key) {
        auto it = lru_nodes_map.find(key);
        if (it == lru_nodes_map.end()){
            return -1;
        }
        // move node to head
        LRUNode* cur_node = it->second;
        _move_lru_node_head(cur_node);
        return cur_node->value;
    }
    
    void put(int key, int value) {
        auto it = lru_nodes_map.find(key);
        if (it != lru_nodes_map.end()){
            LRUNode* cur_node = it->second;
            cur_node->value = value;
            _move_lru_node_head(cur_node);
            return;
        }

        if (m_size == m_capacity) {
            _delete_oldest_node();
        }

        LRUNode* new_node = new LRUNode(key, value);
        new_node->pre = this->head;
        new_node->next = this->head->next;

        this->head->next->pre = new_node;
        this->head->next = new_node;
        this->lru_nodes_map[key] = new_node;
        ++this->m_size;
    }

    void _move_lru_node_head(LRUNode* lru_node_ptr){
        if (lru_node_ptr == nullptr || lru_node_ptr->pre == this->head) {
            return;
        }
        
        // move one lru_node to head
        LRUNode* pre_lru_node = lru_node_ptr->pre;
        LRUNode* next_lru_node = lru_node_ptr->next;

        pre_lru_node->next = next_lru_node;
        next_lru_node->pre = pre_lru_node;

        lru_node_ptr->pre = this->head;
        lru_node_ptr->next = this->head->next;
        this->head->next->pre = lru_node_ptr;
        this->head->next = lru_node_ptr;
    }

    void _delete_oldest_node(){
        if (m_size == 0 || this->tail->pre == this->head) 
            return;

        LRUNode* del_node_ptr = this->tail->pre;
        this->tail->pre = del_node_ptr->pre;
        del_node_ptr->pre->next = this->tail;

        this->lru_nodes_map.erase(del_node_ptr->key);
        delete del_node_ptr;

        --this->m_size;
    }

private:
    int m_capacity = 0;
    int m_size = 0;
    LRUNode* head = nullptr;
    LRUNode* tail = nullptr;
    unordered_map<int, LRUNode*> lru_nodes_map;
};

```

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        // 双指针： 维护一个 [i, j] 窗口
        // 用 map 记录各个 char 最后的位置，遍历一遍即可求解最长子串
        int l = 0;
        int len_longest_substr  = 0;
        unordered_map<char, int> final_index_map;

        for(int r = 0; r < s.size(); ++r){
            char cur_char = s[r];
            if (final_index_map.find(cur_char) != final_index_map.end() 
                && final_index_map[cur_char] >= l
            ){
                l = final_index_map[cur_char] + 1;
            }
            final_index_map[cur_char] = r;
            len_longest_substr = max(len_longest_substr, r-l+1);
        }
        return len_longest_substr;
    }
};
```

``` c++
class Solution {
public:
  int maximalSquare(vector<vector<char>>& matrix) {
      // dp(i,j)=min(dp(i−1,j),dp(i−1,j−1),dp(i,j−1))+1
      if (matrix.size() == 0 || matrix[0].size() == 0) {
          return 0;
      }
      int maxSide = 0;
      int rows = matrix.size(), columns = matrix[0].size();
      vector<vector<int>> dp(rows, vector<int>(columns));
      for (int i = 0; i < rows; i++) {
          for (int j = 0; j < columns; j++) {
              if (matrix[i][j] == '1') {
                  if (i == 0 || j == 0) {
                      dp[i][j] = 1;
                  } else {
                      dp[i][j] = min(min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1;
                  }
                  maxSide = max(maxSide, dp[i][j]);
              }
          }
      }
      int maxSquare = maxSide * maxSide;
      return maxSquare;
  }
};
```

#### 9.1.1 快速排序
<span id="note-215-kestlargest"></span>

```c++
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int n = nums.size();
        return _quickSelect(nums, 0, n-1, n-k);
    }

    int _quickSelect(vector<int> &nums, int l, int r, int k) {
        if (l == r)
            return nums[k];
        int partition = nums[l], i = l - 1, j = r + 1;
        while (i < j) {
            do i++; while (nums[i] < partition);
            do j--; while (nums[j] > partition);
            if (i < j)
                swap(nums[i], nums[j]);
        }
        if (k <= j)return _quickSelect(nums, l, j, k);
        else return _quickSelect(nums, j + 1, r, k);
    }
};
```


### 9.2 二分搜索

<span id="note-33">
旋转排序后，整个数组可以划分为“左”和“右”两个升序子数组，通过 mid 和两端元素（0，n-1）先判断当前 mid 的分区，再根据 target 是否在分区做二分。
</span>

``` c++
int search(vector<int>& nums, int target) {
  int n = (int)nums.size();
  if (!n) {
      return -1;
  }
  if (n == 1) {
      return nums[0] == target ? 0 : -1;
  }
  int l = 0, r = n - 1;

  // 将数组一分为二，其中一个有序，另一个可能有序，在有序的部分上搜索即可
  // 对无序部分再一分为二，同样可以拆分成有序 + 无序
  while (l <= r) {
      int mid = (l + r) / 2;
      if (nums[mid] == target) return mid;
      if (nums[0] <= nums[mid]) {
          if (nums[0] <= target && target < nums[mid]) {
              r = mid - 1;
          } else {
              l = mid + 1;
          }
      } else {
          if (nums[mid] < target && target <= nums[n - 1]) {
              l = mid + 1;
          } else {
              r = mid - 1;
          }
      }
  }
  return -1;
}
```
